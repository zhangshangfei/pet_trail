ALTER TABLE users ADD COLUMN status TINYINT(1) DEFAULT 1 COMMENT '状态: 1-正常 0-已删除';

INNER JOIN follows f ON p.user_id = f.followee_id
INNER JOIN users u ON p.user_id = u.id

AND u.status = 1
